    <script>
        // Initialize AOS
        AOS.init({
            duration: 800,
            once: true,
        });

        // JavaScript para el acordeón de FAQ
        const faqContainer = document.getElementById('faq-container');
        if (faqContainer) {
            faqContainer.addEventListener('click', function(e) {
                const toggle = e.target.closest('.faq-toggle');
                if (toggle) {
                    const content = toggle.nextElementSibling;
                    const icon = toggle.querySelector('svg');
                    
                    const allToggles = faqContainer.querySelectorAll('.faq-toggle');
                    allToggles.forEach(t => {
                        if (t !== toggle) {
                            t.nextElementSibling.classList.add('hidden');
                            t.querySelector('svg').classList.remove('rotate-180');
                        }
                    });

                    content.classList.toggle('hidden');
                    icon.classList.toggle('rotate-180');
                }
            });
        }
        
        // JavaScript para el contador de stock (adaptado de RENOVA+)
        const stockElement = document.getElementById('stock-counter');
        let stockCount = 20; // Stock para la talla S
        const minStock = Math.floor(Math.random() * 5) + 2; // Detiene la cuenta en un número aleatorio entre 2 y 6

        function updateStock() {
            if (stockCount > minStock) {
                stockCount--;
                stockElement.textContent = stockCount;
            } else {
                stockElement.textContent = `¡Últimas ${minStock} unidades!`;
            }
            
            // Establece un nuevo intervalo aleatorio para la próxima actualización
            const randomInterval = Math.random() * 12000 + 8000; // 8-20 segundos
            setTimeout(updateStock, randomInterval);
        }

        // Inicia el contador de stock después de un retraso inicial
        setTimeout(updateStock, 10000); // Primera actualización después de 10 segundos

        // JavaScript para ocultar la flecha de scroll
        const scrollArrow = document.querySelector('.scroll-down-arrow');
        if (scrollArrow) {
